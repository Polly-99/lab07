Создание CMakeLists.txt

1 $ sed -i '/option(BUILD_EXAMPLES "Build examples" OFF)/a\ #добавляем строку в потоковом текстовом редакторе
2 option(BUILD_TESTS "Build tests" OFF)
3 ' CMakeLists.txt
4 $ cat >> CMakeLists.txt <<EOF # редактирование файла CMakeLists.txt
5 if(BUILD_TESTS)
6  enable_testing() # включение тестирования
7  file(GLOB \${PROJECT_NAME}_TEST_SOURCES tests/*.cpp)
8  add_executable(check \${\${PROJECT_NAME}_TEST_SOURCES})
9  target_link_libraries(check \${PROJECT_NAME} \${DEPENDS_LIBRARIES})
10  add_test(NAME check COMMAND check "-s" "-r" "compact" "--use-colour" "yes") # Добавление теста к проекту; -s - вывод успешного выполнения тестов ; -r compact - формат вывода
11 endif()
12 EOF

Создание теста

1 $ cat >> tests/test1.cpp <<EOF
2 #include "catch.hpp"
3 #include <print.hpp>
4 
5 TEST_CASE("output values should match input values", "[file]") {
6  std::string text = "hello";
7  std::ofstream out("file.txt");
8 
9  print(text, out);
10  out.close();
11 
12  std::string result;
13  std::ifstream in("file.txt");
14  in >> result;
15 
16  REQUIRE(result == text);
17 }
18 EOF
